SpigotExtension

open class SpigotExtension(project: Project)(source)

Spigot configuration for the plugin.yml description.

Groovy example:

spigot {
  authors 'Me', 'Someone'
  depends 'ProtocolLib', 'Vault'
  softDepends 'MySoftLib'
  apiVersion '1.15'
  load STARTUP
  commands {
    give {
      aliases 'giv', 'i'
      description 'Give command.'
      permission 'test.foo'
      permissionMessage 'You do not have permission!'
      usage '/<command> [item] [amount]'
    }
  }
  permissions {
    'test.foo' {
      description 'Allows the foo command'
      defaults 'true'
    }
    'test.*' {
      description 'Wildcard permission'
      defaults 'op'
      children = ['test.foo': true]
    }
  }
  debug {
    eula = true
  }
}

Kotlin example:

import io.typst.spigradle.Load

spigot {
  authors = "Me"
  depends = listOf("ProtocolLib", "Vault")
  softDepends = listOf("MySoftLib")
  apiVersion = "1.15"
  load = Load.STARTUP
  commands {
    register("give") {
      aliases = listOf("giv", "i")
      description = "Give command."
      permission = "test.foo"
      permissionMessage = "You do not have permission!"
      usage = "/<command> [item] [amount]"
    }
  }
  permissions {
    register("test.foo") {
      description = "Allows the foo command"
      defaults = "true"
    }
    register("test.*") {
      description = "Wildcard permission"
      defaults = "op"
      children = mapOf("test.foo" to true)
    }
  }
}
