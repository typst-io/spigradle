NukkitExtension

open class NukkitExtension(project: Project)(source)

Nukkit configuration for the 'plugin.yml' description, and debug settings.

Groovy Example:

spigot {
  authors 'Me'
  depends 'ProtocolLib', 'Vault'
  api '1.0.5'
  load STARTUP
  commands {
    give {
      aliases 'giv', 'i'
      description 'Give command.'
      permission 'test.foo'
      permissionMessage 'You do not have the permission!'
      usage '/<command> [item] [amount]'
    }
  }
  permissions {
    'test.foo' {
      description 'Allows foo command'
      defaults 'true'
    }
    'test.*' {
      description 'Wildcard permission'
      defaults 'op'
      children = ['test.foo': true]
    }
  }
  debug {
    eula = true
  }
}

Kotlin Example:

import io.typst.spigradle.Load

spigot {
  authors = "Me"
  depends = listOf("ProtocolLib", "Vault")
  api = listOf("1.0.5")
  load = Load.STARTUP
  commands {
    register("give") {
      aliases = listOf("giv", "i")
      description = "Give command."
      permission = "test.foo"
      permissionMessage = "You do not have the permission!"
      usage = "/<command> [item] [amount]"
    }
  }
  permissions {
    register("test.foo") {
      description = "Allows foo command"
      defaults = "true"
    }
    register("test.*") {
      description = "Wildcard permission"
      defaults = "op"
      children = mapOf("test.foo" to true)
    }
  }
}
